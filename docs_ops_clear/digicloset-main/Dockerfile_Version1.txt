# Generic Dockerfile for a Node.js web app. Adjust as needed.
FROM node:18-alpine

# Create app directory
WORKDIR /usr/src/app

# Install dependencies
COPY package*.json ./
RUN npm ci --only=production

# Copy app source
COPY . .

# If you build a production bundle, uncomment:
# RUN npm run build

# Default port (change to your app port)
EXPOSE 3000

# Start command (adjust if your start script differs)
CMD ["npm", "start"]
